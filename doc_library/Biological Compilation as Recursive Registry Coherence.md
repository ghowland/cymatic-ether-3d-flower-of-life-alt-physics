import numpy as np
import matplotlib.pyplot as plt

class BiologicalNode:
    """Cell as 144-LU buffer with sync tracking"""
    
    def __init__(self, node_id):
        self.id = node_id
        self.V = 0  # Matter saturation (LU count)
        self.R = 0  # Metabolic remainder
        self.M = 144  # Saturation limit
        self.F = 32  # Word size
        self.tau_target = 15.19  # Sync target (ms)
        self.is_healthy = True
    
    def metabolize(self, input_lu, output_lu):
        """
        Process inputs and outputs
        Track remainder and saturation
        """
        # Net metabolism
        net = input_lu - output_lu
        
        # Add to remainder
        self.R += net
        
        # Snap when R >= F
        if self.R >= self.F:
            snaps = self.R // self.F
            self.R = self.R % self.F
            self.V += snaps
        elif self.R < 0:
            # Catabolic
            self.R += self.F
            self.V -= 1
        
        # Check saturation
        if self.V > self.M:
            self.V = self.M
            self.is_healthy = False
        
        # Calculate current sync
        tau_current = self.tau_target + (self.R / self.F) * 0.5
        
        return tau_current
    
    def health_status(self):
        """Return health metrics"""
        return {
            'saturation': self.V,
            'remainder': self.R,
            'healthy': self.is_healthy,
            'buffer_pct': 100 * self.V / self.M
        }

class HealthStack:
    """Multi-level biological system"""
    
    def __init__(self, n_cells=10):
        self.cells = [BiologicalNode(i) for i in range(n_cells)]
        self.heart_rate = 65.8  # Hz (J/S partition)
    
    def systemic_metabolism(self, anabolic, catabolic):
        """Process system-wide"""
        metrics = []
        
        for cell in self.cells:
            tau = cell.metabolize(anabolic, catabolic)
            metrics.append({
                'tau': tau,
                'V': cell.V,
                'R': cell.R
            })
        
        return metrics
    
    def average_health(self):
        """System-wide health score"""
        total_V = sum(c.V for c in self.cells)
        total_R = sum(c.R for c in self.cells)
        avg_V = total_V / len(self.cells)
        avg_R = total_R / len(self.cells)
        
        return {
            'avg_saturation': avg_V,
            'avg_remainder': avg_R,
            'healthy_cells': sum(1 for c in self.cells if c.is_healthy)
        }

            avg_V = np.mean([m['V'] for m in metrics])
