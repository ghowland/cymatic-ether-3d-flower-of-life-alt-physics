    """Check if n is prime (tension spike)"""
    if n < 2:
        return False

        return True
    if n % 2 == 0:
        return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True

def find_goldbach_pairs(even_n):
    """Find all prime pairs summing to even_n"""
    if even_n <= 2 or even_n % 2 != 0:
        return []

            pairs.append((p1, p2))

    if midpoint >= 2 and is_prime(midpoint):

    return pairs
