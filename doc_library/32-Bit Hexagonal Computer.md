def audit_segment(gps_start, duration=4096):
    # 1. Download raw strain data
    data = TimeSeries.fetch_open_data('H1', gps_start, gps_start+duration)
    
    # 2. Preprocessing
    fs = 16384  # Hz
    raw = np.nan_to_num(data.value)

    # 3. High-resolution spectral analysis
    nperseg = fs * 32  # 32-second segments → 0.03125 Hz bins
    f_axis, Pxx = welch(raw, fs, nperseg=nperseg, noverlap=nperseg//2)
    
    # 4. Peak detection in 2-4 Hz band
    mask = (f_axis >= 2.0) & (f_axis <= 4.0)
    peak_idx = np.argmax(Pxx[mask])

    bin_width = 0.03125

    return {
        'gps': gps_start,
        'frequency': f_detected,
        'harmonic': harmonic_n,
        'residual': residual
    }
