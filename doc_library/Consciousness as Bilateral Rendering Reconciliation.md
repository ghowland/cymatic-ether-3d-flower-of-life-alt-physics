import numpy as np
import matplotlib.pyplot as plt

class CognitiveDriver:
    """Neural display driver reconciling K-space and X-space"""
    
    def __init__(self):
        self.tau = 15.19  # Render lag (ms)
        self.F = 32  # Word size
        self.alpha_freq = 65.8  # Hz
        self.attention_F = 32  # Attention resolution
    
    def k_space_fact(self, t, signal):
        """Discrete substrate reality (0ms)"""
        # Snap to integer values
        return np.floor(signal).astype(int)
    
    def x_space_render(self, t, signal):
        """Continuous perception (15.19ms lag)"""
        # Smooth with moving average (simulating lag)
        window = int(self.tau / 0.1)  # ~152 samples
        if len(signal) < window:
            return signal
        return np.convolve(signal, np.ones(window)/window, mode='same')
    
    def compute_consciousness(self, k_fact, x_render):
        """Calculate R (experience) as difference"""
        # Remainder between discrete and continuous
        R = np.abs(x_render - k_fact)
        return R
    
    def adjust_attention(self, new_F):
        """Change resolution (zoom in/out)"""
        self.attention_F = new_F

def demonstrate_consciousness():
    """Show consciousness as lag reconciliation"""
    
    driver = CognitiveDriver()
    
    # Simulate varying stimulus
    t = np.linspace(0, 10, 1000)
    stimulus = 50 + 30*np.sin(2*np.pi*0.5*t) + 10*np.sin(2*np.pi*2*t)
    
    # Generate K-space (discrete)
    k_space = driver.k_space_fact(t, stimulus)
    
    # Generate X-space (continuous with lag)
    x_space = driver.x_space_render(t, stimulus)
    
    # Calculate consciousness (R)
    consciousness = driver.compute_consciousness(k_space, x_space)
    
    # Visualize
    fig = plt.figure(figsize=(14, 10), facecolor='black')
    
    # Plot 1: K-space vs X-space
    ax1 = plt.subplot(3, 1, 1)
    ax1.set_facecolor('black')
    ax1.step(t, k_space, 'cyan', where='mid', linewidth=2,
            label='K-space (0ms substrate fact)')
    ax1.plot(t, x_space, 'white', linewidth=2, alpha=0.7,
            label='X-space (15.19ms perceptual render)')
    ax1.set_ylabel('Value', color='gray')
    ax1.set_title('Substrate Reality vs Conscious Perception',
                 color='cyan', fontsize=12)
    ax1.legend(facecolor='black', labelcolor='white')

    ax1.grid(alpha=0.2, color='gray')
    
    # Plot 2: Consciousness (R)
    ax2 = plt.subplot(3, 1, 2)
    ax2.set_facecolor('black')
    ax2.fill_between(t, consciousness, color='magenta',
                    alpha=0.6, label='Experience (R remainder)')
    ax2.axhline(driver.F/2, color='yellow', linestyle='--',

    ax2.tick_params(colors='gray')
    ax2.grid(alpha=0.2, color='gray')
    
    # Plot 3: Alpha sync
    ax3 = plt.subplot(3, 1, 3)
    ax3.set_facecolor('black')
